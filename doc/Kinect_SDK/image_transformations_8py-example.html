53def image_transformations():
54
55 # Open a device using the "with" syntax.
56 with k4a.Device.open() as device:
57
58 # In order to start capturing frames, need to start the cameras.
59 # The start_cameras() function requires a device configuration which
60 # specifies the modes in which to put the color and depth cameras.
61 # For convenience, the k4a package pre-defines some configurations
62 # for common usage of the Azure Kinect device, but the user can
63 # modify the values to set the device in their preferred modes.
64 device_config = k4a.DEVICE_CONFIG_BGRA32_1080P_WFOV_2X2BINNED_FPS15
65 status = device.start_cameras(device_config)
66 if status != k4a.EStatus.SUCCEEDED:
67 raise IOError("Failed to start cameras.")
68
69 # In order to create a Transformation class, we first need to get
70 # a Calibration instance. Getting a calibration object needs the
71 # depth mode and color camera resolution. Thankfully, this is part
72 # of the device configuration used in the start_cameras() function.
73 calibration = device.get_calibration(
74 depth_mode=device_config.depth_mode,
75 color_resolution=device_config.color_resolution)
76
77 # Create a Transformation object using the calibration object as param.
78 transform = k4a.Transformation(calibration)
79
80 # Get a capture using the "with" syntax.
81 with device.get_capture(-1) as capture:
82
83 color = capture.color
84 depth = capture.depth
85 ir = capture.ir
86
87 # Get a color image but transformed in the depth space.
88 color_transformed = transform.color_image_to_depth_camera(depth, color)
89 plot_images(color, depth, color_transformed)
90
91 # Get a depth image but transformed in the color space.
92 depth_transformed = transform.depth_image_to_color_camera(depth)
93 plot_images(color, depth, depth_transformed, cmap='jet')
94
95 # Get a depth image and custom image but transformed in the color
96 # space. Create a custom image. This must have EImageFormat.CUSTOM8
97 # or EImageFormat.CUSTOM16 as the image_format, so create an
98 # entirely new Image and copy the IR data to that image.
99 ir_custom = k4a.Image.create(
100 k4a.EImageFormat.CUSTOM16,
101 ir.width_pixels,
102 ir.height_pixels,
103 ir.stride_bytes)
104 np.copyto(ir_custom.data, ir.data)
105
106 depth_transformed, ir_transformed = transform.depth_image_to_color_camera_custom(
107 depth, ir_custom, k4a.ETransformInterpolationType.LINEAR, 0)
108 plot_images(color, depth_transformed, ir_transformed, cmap='gray')
109
110 # There is no need to delete resources since Python will take care
111 # of releasing resources in the objects' deleters. To explicitly
112 # delete the images, capture, and device objects, call del on them.
113
114if __name__ == '__main__':
115 image_transformations()
