18# This package is used for displaying the images.
19# It is not part of the k4a package and is not a hard requirement for k4a.
20# Users need to install these packages in order to use this module.
21import matplotlib.pyplot as plt
22
23# This will import all the public symbols into the k4a namespace.
24import k4a
25
26
27def simple_viewer():
28
29 # Open a device using the static function Device.open().
30 device = k4a.Device.open()
31
32 # In order to start capturing frames, need to start the cameras.
33 # The start_cameras() function requires a device configuration which
34 # specifies the modes in which to put the color and depth cameras.
35 # For convenience, the k4a package pre-defines some configurations
36 # for common usage of the Azure Kinect device, but the user can
37 # modify the values to set the device in their preferred modes.
38 device_config = k4a.DEVICE_CONFIG_BGRA32_2160P_WFOV_2X2BINNED_FPS15
39 device.start_cameras(device_config)
40
41 # Get a capture.
42 # The -1 tells the device to wait forever until a capture is available.
43 capture = device.get_capture(-1)
44
45 # Open a matplotlib figure to display images.
46 fig = plt.figure()
47 ax = []
48 ax.append(fig.add_subplot(1, 3, 1, label="Color"))
49 ax.append(fig.add_subplot(1, 3, 2, label="Depth"))
50 ax.append(fig.add_subplot(1, 3, 3, label="IR"))
51
52 # The capture has the following fields that can be read right away:
53 # color : the color image
54 # depth : the depth image
55 # ir : the ir image
56 im = []
57 im.append(ax[0].imshow(capture.color.data))
58 im.append(ax[1].imshow(capture.depth.data, cmap='jet'))
59 im.append(ax[2].imshow(capture.ir.data, cmap='gray'))
60
61 ax[0].title.set_text('Color')
62 ax[1].title.set_text('Depth')
63 ax[2].title.set_text('IR')
64
65 # Note: The data in the images is in BGRA planes, but the matplotlib
66 # library expects them to be in RGBA. This results in an inverted color
67 # display if not properly handled. The user can splice the planes as
68 # appropriate or use opencv which has a function call to transform
69 # BGRA into RGBA.
70
71 while fig is not None:
72
73 # Draw the figure with the images.
74 plt.pause(.1)
75 plt.draw()
76
77 # Get a new capture.
78 capture = device.get_capture(-1)
79 if capture is None:
80 del fig
81 break
82
83 # Update the images in the figures.
84 im[0].set_data(capture.color.data)
85 im[1].set_data(capture.depth.data)
86 im[2].set_data(capture.ir.data)
87
88 # There is no need to delete the capture since Python will take care of
89 # that in the object's deleter.
90
91
92 # There is no need to stop the cameras since the deleter will stop
93 # the cameras, but it's still prudent to do it explicitly.
94 device.stop_cameras()
95
96 # There is no need to delete resources since Python will take care
97 # of releasing resources in the objects' deleters.
98
99if __name__ == '__main__':
100 simple_viewer()
