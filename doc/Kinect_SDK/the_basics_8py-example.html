11# This will import all the public symbols into the k4a namespace.
12import k4a
13
14
15def the_basics():
16
17 # Open a device using the static function Device.open().
18 device = k4a.Device.open()
19 if device is None:
20 exit(-1)
21
22 # Print the device serial number, hardware_version, and
23 # color control capabilities.
24 print(device.serial_number)
25 print(device.hardware_version)
26 print(device.color_ctrl_cap)
27
28 # In order to start capturing frames, need to start the cameras.
29 # The start_cameras() function requires a device configuration which
30 # specifies the modes in which to put the color and depth cameras.
31 # See DeviceConfiguration, EImageFormat, EColorResolution, EDepthMode,
32 # EFramesPerSecond, and EWiredSyncMode.
33 device_config = k4a.DeviceConfiguration(
34 color_format=k4a.EImageFormat.COLOR_BGRA32,
35 color_resolution=k4a.EColorResolution.RES_1080P,
36 depth_mode=k4a.EDepthMode.WFOV_2X2BINNED,
37 camera_fps=k4a.EFramesPerSecond.FPS_15,
38 synchronized_images_only=True,
39 depth_delay_off_color_usec=0,
40 wired_sync_mode=k4a.EWiredSyncMode.STANDALONE,
41 subordinate_delay_off_master_usec=0,
42 disable_streaming_indicator=False)
43
44 status = device.start_cameras(device_config)
45 if status != k4a.EStatus.SUCCEEDED:
46 exit(-1)
47
48 # The IMUs can be started but only after starting the cameras.
49 # Note that it returns an EWaitStatus rather than a EStatus.
50 wait_status = device.start_imu()
51 if wait_status != k4a.EWaitStatus.SUCCEEDED:
52 exit(-1)
53
54 imu_sample = device.get_imu_sample(-1)
55 print(imu_sample)
56
57 # Get a capture.
58 # The -1 tells the device to wait forever until a capture is available.
59 capture = device.get_capture(-1)
60
61 # Print the color, depth, and IR image details and the temperature.
62 print(capture.color)
63 print(capture.depth)
64 print(capture.ir)
65 print(capture.temperature)
66
67 # The capture object has fields for the color, depth, and ir images.
68 # These are container classes; the actual image data is stored as a
69 # numpy ndarray object in a field called data. Users can query these
70 # fields directly.
71 color_image = capture.color
72 color_image_data = color_image.data
73
74 # Get the image width and height in pixels, and stride in bytes.
75 width_pixels = color_image.width_pixels
76 height_pixels = color_image.height_pixels
77 stride_bytes = color_image.stride_bytes
78
79 # There is no need to stop the cameras since the deleter will stop
80 # the cameras, but it's still prudent to do it explicitly.
81 device.stop_cameras()
82 device.stop_imu()
83
84 # There is no need to delete resources since Python will take care
85 # of releasing resources in the objects' deleters. To explicitly
86 # delete the images, capture, and device objects, call del on them.
87
88if __name__ == '__main__':
89 the_basics()
